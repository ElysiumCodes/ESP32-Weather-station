#define PIN 2  // The pin connected to the button
#define POUT 17
void setup() {
    pinMode(PIN, OUTPUT);
    pinMode(POUT, INPUT);
    Serial.begin(115200);  
    digitalWrite(PIN, HIGH);       // Initialize serial communication
}

void loop() {
     // Read the state of the button
    int state= digitalRead(POUT);
    if (state == HIGH) {            // If the button is pressed
        Serial.println("Button Pressed!");  // Print message to serial monitor
    }
}
