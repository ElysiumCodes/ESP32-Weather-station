#define PIN 2
#define POUT 5
void setup(){
pinMode(PIN, INPUT);
pinMode(POUT, OUTPUT);
Serial.begin(115200);
}

void loop(){
  int state=digitalRead(PIN);
  if(state==HIGH){
    digitalWrite(POUT,HIGH);
    Serial.print("Pressed");
    delay(1000);
    digitalWrite(POUT,LOW);
  }
}
